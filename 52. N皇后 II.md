### 52. N皇后 II

返回 N 皇后问题解的个数

``` java
// 八皇后问题的经典解法：回溯
class Solution {
    private int count = 0;
    public int totalNQueens(int n) {
        int[] queens = new int[n];
        nQueens(0, n, queens);
        return count;
    }

    void nQueens(int k, int n,  int[] queens) {
        if (k == n) {
            ++count;
            return;
        }
        for (int i = 0; i < n; i++) {
            int j = 0;
            // 判断是否冲突
            for (; j < k; j++) {
                if (queens[j] == i || 
                    Math.abs(queens[j] - i) == Math.abs(k - j)) {
                    break;
                }
            }
            // 不存在冲突则摆放下一颗皇后
            if (j == k) {
                queens[k] = i;
                nQueens(k + 1, n, queens);
            }
        }
    }
}
```
